<script setup>
import { ref, onMounted, computed, watch } from 'vue';
import {
    Wallet,
    TrendingUp,
    TrendingDown,
    DollarSign,
    Filter,
    RefreshCw,
    Loader2,
    X,
    Eye,
    ChevronDown,
    ChevronUp,
    IndianRupee,
    Calendar,
    User,
    Building2,
    Receipt,
    CreditCard,
    Plus,
    CheckCircle,
    AlertCircle,
    FileText,
    Clock
} from 'lucide-vue-next';

const props = defineProps({
    userId: {
        type: [String, Number],
        default: null
    }
});

const apiBaseUrl = import.meta.env.VITE_BACKEND_API_URL;

const emit = defineEmits(['error']);

// State
const loading = ref(false);
const showModal = ref(false);
const showFilters = ref(false);
const selectedBalance = ref(null);
const showBalanceDetail = ref(false);
const showRepaymentModal = ref(false);
const expandedSections = ref({
    summary: true,
    balances: true
});

// Filters
const filters = ref({
    page: 1,
    limit: 10,
    transaction_type: '',
    user_id: props.userId || '',
    is_settled: '',
    search: '',
    from_date: '',
    to_date: '',
    sort_by: 'transaction_date',
    sort_order: 'DESC'
});

// Data
const balancesData = ref({
    balances: [],
    pagination: {
        totalItems: 0,
        currentPage: 1,
        totalPages: 0,
        limit: 10
    }
});

const summaryData = ref({
    pending: { count: 0, total_amount: '0.00' },
    advance: { count: 0, total_amount: '0.00' },
    repayment: { count: 0, total_amount: '0.00' }
});

// Updated to match new API response format
const userBalanceSummary = ref({
    user_id: null,
    balance_type: 'settled', // 'pending', 'advance', or 'settled'
    balance_amount: 0, // Absolute value
    display_message: 'No pending balance',
    pending_transactions: 0,
    advance_transactions: 0,
    repayment_transactions: 0,
    raw_balance: 0 // Keep for internal calculations
});

// Repayment form
const repaymentForm = ref({
    user_id: null,
    repayment_amount: '',
    repaid_by: null,
    repayment_notes: '',
    payment_reference: ''
});

// Updated computed properties for new API format
const currentBalanceColor = computed(() => {
    const type = userBalanceSummary.value.balance_type;
    if (type === 'pending') return 'text-red-600'; // User owes money
    if (type === 'advance') return 'text-green-600'; // User has advance
    return 'text-gray-600'; // Settled
});

const balanceStatusBadge = computed(() => {
    const type = userBalanceSummary.value.balance_type;
    const badges = {
        pending: 'bg-red-100 text-red-700 border-red-300',
        advance: 'bg-green-100 text-green-700 border-green-300',
        settled: 'bg-gray-100 text-gray-700 border-gray-300'
    };
    return badges[type] || badges.settled;
});

const totalTransactions = computed(() => {
    return summaryData.value.pending.count +
        summaryData.value.advance.count +
        summaryData.value.repayment.count;
});

// Methods
const formatCurrency = (value) => {
    if (!value && value !== 0) return '0.00';
    return new Intl.NumberFormat('en-IN', {
        minimumFractionDigits: 2,
        maximumFractionDigits: 2
    }).format(Math.abs(value));
};

const formatDate = (date) => {
    if (!date) return '-';
    return new Date(date).toLocaleDateString('en-IN', {
        day: '2-digit',
        month: 'short',
        year: 'numeric',
        hour: '2-digit',
        minute: '2-digit'
    });
};

const getTransactionTypeColor = (type) => {
    const colors = {
        pending: 'bg-orange-100 text-orange-700 border-orange-300',
        advance: 'bg-blue-100 text-blue-700 border-blue-300',
        repayment: 'bg-purple-100 text-purple-700 border-purple-300'
    };
    return colors[type] || colors.pending;
};

const getAmountTypeColor = (type) => {
    const colors = {
        credit: 'bg-red-100 text-red-700 border-red-300',
        debit: 'bg-green-100 text-green-700 border-green-300'
    };
    return colors[type] || colors.credit;
};

const buildQueryParams = () => {
    const params = { ...filters.value };
    if (props.userId) {
        params.user_id = props.userId;
    }
    const filtered = Object.entries(params).filter(([_, v]) => v !== '');
    return new URLSearchParams(filtered).toString();
};

const fetchSummary = async () => {
    try {
        const params = new URLSearchParams({
            is_settled: filters.value.is_settled,
            ...(filters.value.from_date && { from_date: filters.value.from_date }),
            ...(filters.value.to_date && { to_date: filters.value.to_date })
        });

        const response = await fetch(`${apiBaseUrl}/payment-balances/summary?${params}`);
        const result = await response.json();

        if (result.success && result.data) {
            summaryData.value = result.data;
        }
    } catch (error) {
        console.error('Error fetching summary:', error);
        emit('error', { type: 'fetchSummary', error });
    }
};

const fetchUserBalanceSummary = async () => {
    if (!props.userId) return;

    try {
        const response = await fetch(`${apiBaseUrl}/payment-balances/user/${props.userId}/summary`);
        const result = await response.json();

        if (result.success && result.data) {
            userBalanceSummary.value = result.data;
        }
    } catch (error) {
        console.error('Error fetching user balance summary:', error);
        emit('error', { type: 'fetchUserBalance', error });
    }
};

const fetchBalances = async () => {
    try {
        loading.value = true;
        const params = buildQueryParams();
        const response = await fetch(`${apiBaseUrl}/payment-balances?${params}`);
        const result = await response.json();

        if (result.success && result.data) {
            balancesData.value = result.data;
        }
    } catch (error) {
        console.error('Error fetching balances:', error);
        emit('error', { type: 'fetchBalances', error });
    } finally {
        loading.value = false;
    }
};

const fetchAllData = async () => {
    await Promise.all([
        fetchSummary(),
        props.userId ? fetchUserBalanceSummary() : Promise.resolve(),
        fetchBalances()
    ]);
};

const applyFilters = () => {
    filters.value.page = 1;
    fetchAllData();
};

const resetFilters = () => {
    filters.value = {
        page: 1,
        limit: 10,
        transaction_type: '',
        user_id: props.userId || '',
        is_settled: '',
        search: '',
        from_date: '',
        to_date: '',
        sort_by: 'transaction_date',
        sort_order: 'DESC'
    };
    applyFilters();
};

const changePage = (page) => {
    filters.value.page = page;
    fetchBalances();
};

const viewBalanceDetails = (balance) => {
    selectedBalance.value = balance;
    showBalanceDetail.value = true;
};

const closeBalanceDetail = () => {
    showBalanceDetail.value = false;
    selectedBalance.value = null;
};

const openRepaymentModal = () => {
    repaymentForm.value = {
        user_id: props.userId || null,
        repayment_amount: '',
        repaid_by: null,
        repayment_notes: '',
        payment_reference: `REPAY_${Date.now()}`
    };
    showRepaymentModal.value = true;
};

const createRepayment = async () => {
    try {
        if (!repaymentForm.value.user_id || !repaymentForm.value.repayment_amount) {
            alert('User ID and repayment amount are required');
            return;
        }

        // Validate repayment amount against pending balance
        if (userBalanceSummary.value.balance_type !== 'pending') {
            alert('User has no pending balance to repay');
            return;
        }

        if (parseFloat(repaymentForm.value.repayment_amount) > userBalanceSummary.value.balance_amount) {
            alert(`Repayment amount cannot exceed pending balance of ₹${formatCurrency(userBalanceSummary.value.balance_amount)}`);
            return;
        }

        loading.value = true;
        const response = await fetch(`${apiBaseUrl}/payment-balances/repayment`, {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify(repaymentForm.value)
        });

        const result = await response.json();

        if (result.success) {
            showRepaymentModal.value = false;
            await fetchAllData();
            alert('Repayment created successfully');
        } else {
            alert(result.message || 'Failed to create repayment');
        }
    } catch (error) {
        console.error('Error creating repayment:', error);
        alert('Failed to create repayment');
    } finally {
        loading.value = false;
    }
};

const toggleSection = (section) => {
    expandedSections.value[section] = !expandedSections.value[section];
};

const openModal = () => {
    showModal.value = true;
    document.body.style.overflow = 'hidden';
    fetchAllData();
};

const closeModal = () => {
    showModal.value = false;
    document.body.style.overflow = '';
};

const viewSummaryDetails = (type) => {
    filters.value.transaction_type = type;
    filters.value.page = 1;
    fetchBalances();
};

// Lifecycle
onMounted(() => {
    if (props.userId) {
        fetchUserBalanceSummary();
    }
    fetchSummary();
});

watch(() => props.userId, (newVal) => {
    if (newVal) {
        filters.value.user_id = newVal;
        fetchUserBalanceSummary();
        fetchSummary();
    }
});
</script>

    <!-- Main Card UI -->

                            </div>
                            <div>
                                <h2 class="text-2xl font-bold text-white">Payment Balances</h2>
                                <p class="text-white/80 text-sm mt-0.5">
                                    {{ userId ? 'User Specific' : 'All Users' }}
                                </p>
                            </div>

                            <!-- Only show repayment button if user has pending balance -->

                                <Plus class="w-5 h-5" />
                                Create Repayment

                            <button @click="closeModal" class="p-2 hover:bg-white/20 rounded-lg transition text-white">
                                <X class="w-6 h-6" />
                            </button>
                        </div>
                    </div>

                    <!-- Modal Body -->
                    <div class="flex-1 overflow-y-auto p-6 bg-gray-50">
                        <!-- Loading State -->
                        <div v-if="loading && balancesData.balances.length === 0"
                            class="flex justify-center items-center py-12">
                            <Loader2 class="w-8 h-8 animate-spin text-[#ff5100]" />
                            <span class="ml-3 text-gray-600 font-medium">Loading payment balances...</span>
                        </div>

                        <!-- Content -->
                        <div v-else>
                            <!-- User Balance Card - Updated for new API format -->
                            <div v-if="userId" class="bg-white rounded-xl shadow-sm p-6 mb-6 border border-gray-200">
                                <h3 class="text-lg font-semibold text-gray-900 mb-4 flex items-center gap-2">
                                    <User class="w-5 h-5 text-[#ff5100]" />
                                    User Balance Summary
                                </h3>
                                <div class="grid grid-cols-1 md:grid-cols-4 gap-4">
                                    <div class="bg-gradient-to-br from-indigo-50 to-indigo-100 rounded-lg p-4">
                                        <p class="text-xs text-indigo-700 mb-1">Current Balance</p>
                                        <!-- Display absolute value with proper label -->
                                        <p :class="currentBalanceColor" class="text-2xl font-bold">
                                            ₹{{ formatCurrency(userBalanceSummary.balance_amount) }}
                                        </p>
                                        <span :class="balanceStatusBadge"
                                            class="inline-block mt-2 px-2 py-1 text-xs rounded-full border font-semibold">
                                            <!-- Display admin-friendly label -->
                                            <span v-if="userBalanceSummary.balance_type === 'pending'">User Owes</span>
                                            <span v-else-if="userBalanceSummary.balance_type === 'advance'">User
                                                Advance</span>
                                            <span v-else>Settled</span>
                                        </span>
                                        <!-- Optional: Show display message from API -->
                                        <p class="text-xs text-indigo-600 mt-2">{{ userBalanceSummary.display_message }}
                                        </p>

                                        <p class="text-xs text-orange-600 mt-2">Transactions</p>

                                    <div class="bg-gradient-to-br from-blue-50 to-blue-100 rounded-lg p-4">
                                        <p class="text-xs text-blue-700 mb-1">Advance</p>
                                        <p class="text-2xl font-bold text-blue-900">{{
                                            userBalanceSummary.advance_transactions }}</p>
                                        <p class="text-xs text-blue-600 mt-2">Transactions</p>
                                    </div>
                                    <div class="bg-gradient-to-br from-purple-50 to-purple-100 rounded-lg p-4">
                                        <p class="text-xs text-purple-700 mb-1">Repayments</p>
                                        <p class="text-2xl font-bold text-purple-900">{{
                                            userBalanceSummary.repayment_transactions }}</p>
                                        <p class="text-xs text-purple-600 mt-2">Transactions</p>
                                    </div>
                                </div>
                            </div>

                            <!-- Filters -->
                            <div class="bg-white rounded-xl shadow-sm mb-6 border border-gray-200">
                                <div class="p-4 border-b border-gray-200 flex items-center justify-between">
                                    <h3 class="text-lg font-semibold text-gray-900">Filters</h3>
                                    <button @click="showFilters = !showFilters"
                                        class="flex items-center gap-2 px-4 py-2 text-sm font-medium text-gray-700 bg-gray-100 rounded-lg hover:bg-gray-200 transition">
                                        <Filter class="w-4 h-4" />
                                        {{ showFilters ? 'Hide' : 'Show' }} Filters
                                    </button>
                                </div>
                                <transition enter-active-class="transition-all duration-300"
                                    leave-active-class="transition-all duration-200"
                                    enter-from-class="opacity-0 max-h-0" enter-to-class="opacity-100 max-h-96"
                                    leave-from-class="opacity-100 max-h-96" leave-to-class="opacity-0 max-h-0">
                                    <div v-show="showFilters" class="p-6 bg-gray-50">
                                        <div class="grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-4 gap-4">
                                            <div>
                                                <label class="block text-sm font-medium text-gray-700 mb-2">Transaction
                                                    Type</label>
                                                <select v-model="filters.transaction_type"
                                                    class="w-full px-3 py-2 border border-gray-300 rounded-lg focus:outline-none focus:ring-2 focus:ring-[#ff5100] focus:border-[#ff5100]">
                                                    <option value="">All Types</option>
                                                    <option value="pending">Pending</option>
                                                    <option value="advance">Advance</option>
                                                    <option value="repayment">Repayment</option>
                                                </select>
                                            </div>
                                            <div>
                                                <label class="block text-sm font-medium text-gray-700 mb-2">Settlement
                                                    Status</label>
                                                <select v-model="filters.is_settled"
                                                    class="w-full px-3 py-2 border border-gray-300 rounded-lg focus:outline-none focus:ring-2 focus:ring-[#ff5100] focus:border-[#ff5100]">
                                                    <option value="">All</option>
                                                    <option value="true">Settled</option>
                                                    <option value="false">Unsettled</option>
                                                </select>
                                            </div>
                                            <div>
                                                <label
                                                    class="block text-sm font-medium text-gray-700 mb-2">Search</label>
                                                <input type="text" v-model="filters.search"
                                                    placeholder="Invoice, Customer..."
                                                    class="w-full px-3 py-2 border border-gray-300 rounded-lg focus:outline-none focus:ring-2 focus:ring-[#ff5100] focus:border-[#ff5100]" />
                                            </div>
                                            <div v-if="!userId">
                                                <label class="block text-sm font-medium text-gray-700 mb-2">User
                                                    ID</label>
                                                <input type="number" v-model="filters.user_id"
                                                    placeholder="Filter by user"
                                                    class="w-full px-3 py-2 border border-gray-300 rounded-lg focus:outline-none focus:ring-2 focus:ring-[#ff5100] focus:border-[#ff5100]" />
                                            </div>

                                                <label class="block text-sm font-medium text-gray-700 mb-2">Start
                                                    Date</label>

                                            </div>
                                            <div class="lg:col-span-2 flex items-end gap-2">
                                                <button @click="applyFilters"

                                                    Apply Filters
                                                </button>
                                                <button @click="resetFilters"

                                                </button>

                            <!-- Summary Cards (for all users view) -->

                                <div class="flex items-center justify-between mb-4">
                                    <h3 class="text-lg font-semibold text-gray-900">Summary Overview</h3>

                                        <ChevronDown v-if="expandedSections.summary" class="w-5 h-5" />
                                        <ChevronUp v-else class="w-5 h-5" />
                                    </button>
                                </div>
                                <transition enter-active-class="transition-all duration-300"
                                    leave-active-class="transition-all duration-200"
                                    enter-from-class="opacity-0 -translate-y-2"
                                    enter-to-class="opacity-100 translate-y-0"

                                    <div v-show="expandedSections.summary"

                                        <!-- Pending -->
                                        <div @click="viewSummaryDetails('pending')"
                                            class="bg-gradient-to-br from-orange-50 to-orange-100 rounded-xl p-6 border border-orange-200 cursor-pointer hover:shadow-lg transition">
                                            <div class="flex items-center justify-between mb-3">

                                            <div class="flex items-center justify-between mb-3">
                                                <div class="bg-purple-500 rounded-lg p-2.5">
