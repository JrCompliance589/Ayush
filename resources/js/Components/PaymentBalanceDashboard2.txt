<script setup>
import { ref, onMounted, computed, watch } from 'vue';
import {
    Wallet,
    TrendingDown,
    Filter,
    RefreshCw,
    Loader2,
    X,
    Eye,
    ChevronDown,
    ChevronUp,
    IndianRupee,
    User,
    Building2,
    Receipt,
    CreditCard,
    Plus,
    CheckCircle,
    AlertCircle,
    FileText,
    Clock,
    Users,
    ArrowUpCircle,
    ArrowDownCircle,
    MinusCircle
} from 'lucide-vue-next';

const props = defineProps({
    userId: {
        type: [String, Number],
        default: null
    }
});

const apiBaseUrl = import.meta.env.VITE_BACKEND_API_URL;
const emit = defineEmits(['error']);

// State Management
const loading = ref(false);
const showModal = ref(false);
const showFilters = ref(false);
const selectedBalance = ref(null);
const showBalanceDetail = ref(false);
const showRepaymentModal = ref(false);
const showUsersWithBalances = ref(false);
const expandedSections = ref({
    summary: true,
    balances: true,
    users: true
});

// Filters
const filters = ref({
    page: 1,
    limit: 10,
    transaction_type: '',
    user_id: props.userId || '',
    is_settled: '',
    search: '',
    from_date: '',
    to_date: '',
    sort_by: 'transaction_date',
    sort_order: 'DESC'
});

// Users with balances filters
const usersFilters = ref({
    page: 1,
    limit: 50,
    balance_type: ''
});

// Data
const balancesData = ref({
    balances: [],
    pagination: {
        totalItems: 0,
        currentPage: 1,
        totalPages: 0,
        limit: 10
    }
});

const summaryData = ref({
    pending: { count: 0, total_amount: '0.00', accounting_type: 'debit' },
    advance: { count: 0, total_amount: '0.00', accounting_type: 'credit' },
    repayment: { count: 0, total_amount: '0.00', accounting_type: 'debit' }
});

const userBalanceSummary = ref({
    user_id: null,
    balance_type: 'settled',
    balance_amount: 0,
    accounting_type: null,
    display_message: 'No pending balance',
    pending_transactions: 0,
    advance_transactions: 0,
    repayment_transactions: 0,
    raw_balance: 0
});

const usersWithBalancesData = ref({
    users: [],
    pagination: {
        totalItems: 0,
        currentPage: 1,
        totalPages: 0,
        limit: 50
    },
    summary: {
        total_users_with_balance: 0,
        total_pending_amount: '0.00',
        total_advance_amount: '0.00'
    }
});

// Repayment form
const repaymentForm = ref({
    user_id: null,
    repayment_amount: '',
    repaid_by: null,
    repayment_notes: '',
    payment_reference: ''
});

// Computed Properties
const currentBalanceColor = computed(() => {
    const type = userBalanceSummary.value.balance_type;
    if (type === 'pending') return 'text-orange-600';
    if (type === 'advance') return 'text-green-600';
    return 'text-gray-600';
});

const balanceStatusBadge = computed(() => {
    const type = userBalanceSummary.value.balance_type;
    const badges = {
        pending: 'bg-orange-100 text-orange-700 border-orange-300',
        advance: 'bg-green-100 text-green-700 border-green-300',
        settled: 'bg-gray-100 text-gray-700 border-gray-300'
    };
    return badges[type] || badges.settled;
});

const totalTransactions = computed(() => {
    return summaryData.value.pending.count +
        summaryData.value.advance.count +
        summaryData.value.repayment.count;
});

const balanceIcon = computed(() => {
    const type = userBalanceSummary.value.balance_type;
    if (type === 'pending') return ArrowDownCircle;
    if (type === 'advance') return ArrowUpCircle;
    return MinusCircle;
});

// Utility Methods
const formatCurrency = (value) => {
    if (!value && value !== 0) return '0.00';
    return new Intl.NumberFormat('en-IN', {
        minimumFractionDigits: 2,
        maximumFractionDigits: 2
    }).format(Math.abs(value));
};

const formatDate = (date) => {
    if (!date) return '-';
    return new Date(date).toLocaleDateString('en-IN', {
        day: '2-digit',
        month: 'short',
        year: 'numeric',
        hour: '2-digit',
        minute: '2-digit'
    });
};

const getTransactionTypeColor = (type) => {
    const colors = {
        pending: 'bg-orange-100 text-orange-700 border-orange-300',
        advance: 'bg-green-100 text-green-700 border-green-300',
        repayment: 'bg-purple-100 text-purple-700 border-purple-300'
    };
    return colors[type] || colors.pending;
};

const getTransactionTypeIcon = (type) => {
    const icons = {
        pending: ArrowDownCircle,
        advance: ArrowUpCircle,
        repayment: RefreshCw
    };
    return icons[type] || Clock;
};

const getBalanceTypeLabel = (type) => {
    const labels = {
        pending: 'Pending',
        advance: 'Advance',
        settled: 'Settled'
    };
    return labels[type] || 'Unknown';
};

const buildQueryParams = () => {
    const params = { ...filters.value };
    if (props.userId) {
        params.user_id = props.userId;
    }
    const filtered = Object.entries(params).filter(([_, v]) => v !== '');
    return new URLSearchParams(filtered).toString();
};

// API Methods
const fetchSummary = async () => {
    try {
        const params = new URLSearchParams({
            is_settled: filters.value.is_settled || 'false',
            ...(filters.value.from_date && { from_date: filters.value.from_date }),
            ...(filters.value.to_date && { to_date: filters.value.to_date }),
            ...(filters.value.user_id && { user_id: filters.value.user_id })
        });

        const response = await fetch(`${apiBaseUrl}/payment-balances/summary?${params}`);
        const result = await response.json();

        if (result.success && result.data) {
            summaryData.value = result.data;
        }
    } catch (error) {
        console.error('Error fetching summary:', error);
        emit('error', { type: 'fetchSummary', error });
    }
};

const fetchUserBalanceSummary = async () => {
    if (!props.userId) return;

    try {
        const response = await fetch(`${apiBaseUrl}/payment-balances/user/${props.userId}/summary`);
        const result = await response.json();

        if (result.success && result.data) {
            userBalanceSummary.value = result.data;
        }
    } catch (error) {
        console.error('Error fetching user balance summary:', error);
        emit('error', { type: 'fetchUserBalance', error });
    }
};

const fetchBalances = async () => {
    try {
        loading.value = true;
        const params = buildQueryParams();
        const response = await fetch(`${apiBaseUrl}/payment-balances?${params}`);
        const result = await response.json();

        if (result.success && result.data) {
            balancesData.value = result.data;
        }
    } catch (error) {
        console.error('Error fetching balances:', error);
        emit('error', { type: 'fetchBalances', error });
    } finally {
        loading.value = false;
    }
};

const fetchUsersWithBalances = async () => {
    try {
        const params = new URLSearchParams({
            page: usersFilters.value.page,
            limit: usersFilters.value.limit,
            ...(usersFilters.value.balance_type && { balance_type: usersFilters.value.balance_type })
        });

        const response = await fetch(`${apiBaseUrl}/payment-balances/users-with-balances?${params}`);
        const result = await response.json();

        if (result.success && result.data) {
            usersWithBalancesData.value = result.data;
        }
    } catch (error) {
        console.error('Error fetching users with balances:', error);
        emit('error', { type: 'fetchUsersWithBalances', error });
    }
};

const fetchAllData = async () => {
    await Promise.all([
        fetchSummary(),
        props.userId ? fetchUserBalanceSummary() : Promise.resolve(),
        fetchBalances(),
        !props.userId ? fetchUsersWithBalances() : Promise.resolve()
    ]);
};

// Filter Actions
const applyFilters = () => {
    filters.value.page = 1;
    fetchAllData();
};

const resetFilters = () => {
    filters.value = {
        page: 1,
        limit: 10,
        transaction_type: '',
        user_id: props.userId || '',
        is_settled: '',
        search: '',
        from_date: '',
        to_date: '',
        sort_by: 'transaction_date',
        sort_order: 'DESC'
    };
    applyFilters();
};

const changePage = (page) => {
    filters.value.page = page;
    fetchBalances();
};

const changeUsersPage = (page) => {
    usersFilters.value.page = page;
    fetchUsersWithBalances();
};

const filterUsersByBalanceType = (type) => {
    usersFilters.value.balance_type = type;
    usersFilters.value.page = 1;
    fetchUsersWithBalances();
};

// Modal Actions
const viewBalanceDetails = (balance) => {
    selectedBalance.value = balance;
    showBalanceDetail.value = true;
};

const closeBalanceDetail = () => {
    showBalanceDetail.value = false;
    selectedBalance.value = null;
};

const openRepaymentModal = (userId = null) => {
    repaymentForm.value = {
        user_id: userId || props.userId || null,
        repayment_amount: '',
        repaid_by: null,
        repayment_notes: '',
        payment_reference: `REPAY_${Date.now()}`
    };
    showRepaymentModal.value = true;
};

const createRepayment = async () => {
    try {
        if (!repaymentForm.value.user_id || !repaymentForm.value.repayment_amount) {
            alert('User ID and repayment amount are required');
            return;
        }

        if (!repaymentForm.value.repaid_by) {
            alert('Repaid by (admin user ID) is required');
            return;
        }

        loading.value = true;
        const response = await fetch(`${apiBaseUrl}/payment-balances/repayment`, {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify(repaymentForm.value)
        });

        const result = await response.json();

        if (result.success) {
            showRepaymentModal.value = false;
            await fetchAllData();
            alert('Repayment created successfully');
        } else {
            alert(result.message || 'Failed to create repayment');
        }
    } catch (error) {
        console.error('Error creating repayment:', error);
        alert('Failed to create repayment: ' + error.message);
    } finally {
        loading.value = false;
    }
};

const toggleSection = (section) => {
    expandedSections.value[section] = !expandedSections.value[section];
};

const openModal = () => {
    showModal.value = true;
    document.body.style.overflow = 'hidden';
    fetchAllData();
};

const closeModal = () => {
    showModal.value = false;
    document.body.style.overflow = '';
};

const viewSummaryDetails = (type) => {
    filters.value.transaction_type = type;
    filters.value.page = 1;
    fetchBalances();
};

const viewUserDetails = (userId) => {
    filters.value.user_id = userId;
    filters.value.page = 1;
    fetchBalances();
    showUsersWithBalances.value = false;
};

// Lifecycle
onMounted(() => {
    if (props.userId) {
        fetchUserBalanceSummary();
    }
    fetchSummary();
});

watch(() => props.userId, (newVal) => {
    if (newVal) {
        filters.value.user_id = newVal;
        fetchUserBalanceSummary();
        fetchSummary();
    }
});
</script>

<template>
    <!-- Main Card UI -->
    <div class="relative bg-gradient-to-br from-orange-50 to-orange-100 rounded-lg p-4 border border-orange-300 hover:shadow-lg transition-shadow duration-200">
        <div class="flex items-center justify-between mb-2">
            <div class="bg-[#ff5100] rounded-lg p-2">
                <Wallet class="w-4 h-4 text-white" />
            </div>
            <component :is="balanceIcon" class="w-4 h-4" :class="currentBalanceColor" />
        </div>
        <p class="text-xs text-gray-700 font-medium mb-1">Payment Balance</p>
        
        <p v-if="userId" :class="currentBalanceColor" class="text-lg font-bold flex items-center gap-1">
            <span v-if="userBalanceSummary.balance_type === 'pending'">-</span>
            <span v-else-if="userBalanceSummary.balance_type === 'advance'">+</span>
            ₹{{ formatCurrency(userBalanceSummary.balance_amount) }}
        </p>
        <p v-else class="text-lg font-bold text-gray-900">
            ₹{{ formatCurrency(summaryData.pending.total_amount) }}
        </p>
        
        <p v-if="userId" class="text-xs mt-1" :class="currentBalanceColor">
            <span v-if="userBalanceSummary.balance_type === 'pending'">Pending Payment</span>
            <span v-else-if="userBalanceSummary.balance_type === 'advance'">Advance Payment</span>
            <span v-else>Settled</span>
        </p>
        <p v-else class="text-xs text-gray-600 mt-1">{{ totalTransactions }} transactions</p>

        <div class="flex items-center justify-between absolute bottom-4 right-4">
            <button @click="openModal"
                class="px-3 py-1.5 bg-[#ff5100] text-white text-xs rounded-md hover:bg-[#e64900] transition font-medium flex items-center gap-1.5 shadow-sm hover:shadow-md">
                <Eye class="w-3.5 h-3.5" />
                View Details
            </button>
        </div>
    </div>

    <!-- Full Details Modal -->
    <transition enter-active-class="transition-opacity duration-300"
        leave-active-class="transition-opacity duration-200" enter-from-class="opacity-0" enter-to-class="opacity-100"
        leave-from-class="opacity-100" leave-to-class="opacity-0">
        <div v-if="showModal" class="fixed inset-0 bg-black bg-opacity-50 z-50 flex items-center justify-center p-4"
            @click.self="closeModal">
            <transition enter-active-class="transition-all duration-300 ease-out"
                leave-active-class="transition-all duration-200 ease-in" enter-from-class="opacity-0 scale-95"
                enter-to-class="opacity-100 scale-100" leave-from-class="opacity-100 scale-100"
                leave-to-class="opacity-0 scale-95">
                <div v-if="showModal"
                    class="bg-white rounded-xl shadow-2xl w-full max-w-7xl max-h-[90vh] overflow-hidden flex flex-col">
                    <!-- Modal Header -->
                    <div
                        class="flex flex-col sm:flex-row items-start sm:items-center justify-between p-4 sm:p-6 border-b border-gray-200 bg-gradient-to-r from-[#ff5100] to-[#ff7340] gap-3">
                        <div class="flex items-center gap-3">
                            <div class="bg-white/20 backdrop-blur-sm rounded-lg p-2">
                                <Wallet class="w-6 h-6 text-white" />
                            </div>
                            <div>
                                <h2 class="text-xl sm:text-2xl font-bold text-white">Payment Balances</h2>
                                <p class="text-white/80 text-sm mt-0.5">
                                    {{ userId ? `User #${userId}` : 'All Users Dashboard' }}
                                </p>
                            </div>
                        </div>
                        <div class="flex items-center gap-2 sm:gap-3 w-full sm:w-auto">
                            <button v-if="userId && userBalanceSummary.balance_type === 'pending'"
                                @click="openRepaymentModal()"
                                class="flex-1 sm:flex-none px-3 sm:px-4 py-2 bg-white text-[#ff5100] rounded-lg hover:bg-gray-100 transition font-medium flex items-center justify-center gap-2 text-sm">
                                <Plus class="w-4 h-4" />
                                <span class="hidden sm:inline">Repayment</span>
                            </button>
                            <button @click="closeModal" class="p-2 hover:bg-white/20 rounded-lg transition text-white">
                                <X class="w-5 h-5 sm:w-6 sm:h-6" />
                            </button>
                        </div>
                    </div>

                    <!-- Modal Body -->
                    <div class="flex-1 overflow-y-auto p-4 sm:p-6 bg-gray-50">
                        <div v-if="loading && balancesData.balances.length === 0"
                            class="flex justify-center items-center py-12">
                            <Loader2 class="w-8 h-8 animate-spin text-[#ff5100]" />
                            <span class="ml-3 text-gray-600 font-medium">Loading payment balances...</span>
                        </div>

                        <div v-else>
                            <!-- User Balance Summary Card -->
                            <div v-if="userId" class="bg-white rounded-xl shadow-sm p-4 sm:p-6 mb-6 border border-gray-200">
                                <h3 class="text-base sm:text-lg font-semibold text-gray-900 mb-4 flex items-center gap-2">
                                    <User class="w-5 h-5 text-[#ff5100]" />
                                    User Balance Summary
                                </h3>
                                <div class="grid grid-cols-2 lg:grid-cols-4 gap-3 sm:gap-4">
                                    <div class="col-span-2 lg:col-span-1 bg-gradient-to-br from-orange-50 to-orange-100 rounded-lg p-4">
                                        <p class="text-xs text-orange-700 mb-1">Current Balance</p>
                                        <p :class="currentBalanceColor" class="text-xl sm:text-2xl font-bold flex items-center gap-1">
                                            <span v-if="userBalanceSummary.balance_type === 'pending'">-</span>
                                            <span v-else-if="userBalanceSummary.balance_type === 'advance'">+</span>
                                            ₹{{ formatCurrency(userBalanceSummary.balance_amount) }}
                                        </p>
                                        <span :class="balanceStatusBadge"
                                            class="inline-block mt-2 px-2 py-1 text-xs rounded-full border font-semibold">
                                            {{ getBalanceTypeLabel(userBalanceSummary.balance_type) }}
                                        </span>
                                    </div>
                                    <div class="bg-gradient-to-br from-orange-50 to-orange-100 rounded-lg p-4">
                                        <p class="text-xs text-orange-700 mb-1">Pending</p>
                                        <p class="text-xl sm:text-2xl font-bold text-orange-900">{{
                                            userBalanceSummary.pending_transactions }}</p>
                                        <p class="text-xs text-orange-600 mt-2">Transactions</p>
                                    </div>
                                    <div class="bg-gradient-to-br from-green-50 to-green-100 rounded-lg p-4">
                                        <p class="text-xs text-green-700 mb-1">Advance</p>
                                        <p class="text-xl sm:text-2xl font-bold text-green-900">{{
                                            userBalanceSummary.advance_transactions }}</p>
                                        <p class="text-xs text-green-600 mt-2">Transactions</p>
                                    </div>
                                    <div class="bg-gradient-to-br from-purple-50 to-purple-100 rounded-lg p-4">
                                        <p class="text-xs text-purple-700 mb-1">Repayments</p>
                                        <p class="text-xl sm:text-2xl font-bold text-purple-900">{{
                                            userBalanceSummary.repayment_transactions }}</p>
                                        <p class="text-xs text-purple-600 mt-2">Transactions</p>
                                    </div>
                                </div>
                            </div>

                            <!-- Users with Balances (Only for All Users View) -->
                            <div v-if="!userId" class="mb-6">
                                <div class="flex items-center justify-between mb-4">
                                    <h3 class="text-base sm:text-lg font-semibold text-gray-900 flex items-center gap-2">
                                        <Users class="w-5 h-5 text-[#ff5100]" />
                                        Users with Outstanding Balances
                                    </h3>
                                    <button @click="toggleSection('users')" class="text-gray-500 hover:text-gray-700">
                                        <ChevronDown v-if="expandedSections.users" class="w-5 h-5" />
                                        <ChevronUp v-else class="w-5 h-5" />
                                    </button>
                                </div>
                                <transition enter-active-class="transition-all duration-300"
                                    leave-active-class="transition-all duration-200">
                                    <div v-show="expandedSections.users">
                                        <!-- Summary Stats -->
                                        <div class="grid grid-cols-1 sm:grid-cols-3 gap-3 sm:gap-4 mb-4">
                                            <div @click="filterUsersByBalanceType('')"
                                                class="bg-gradient-to-br from-indigo-50 to-indigo-100 rounded-lg p-4 cursor-pointer hover:shadow-md transition border border-indigo-200">
                                                <p class="text-xs text-indigo-700 mb-1">Total Users</p>
                                                <p class="text-xl sm:text-2xl font-bold text-indigo-900">{{
                                                    usersWithBalancesData.summary.total_users_with_balance }}</p>
                                            </div>
                                            <div @click="filterUsersByBalanceType('pending')"
                                                class="bg-gradient-to-br from-orange-50 to-orange-100 rounded-lg p-4 cursor-pointer hover:shadow-md transition border border-orange-200">
                                                <p class="text-xs text-orange-700 mb-1 flex items-center gap-1">
                                                    <ArrowDownCircle class="w-3 h-3" />
                                                    Total Pending
                                                </p>
                                                <p class="text-xl sm:text-2xl font-bold text-orange-900">₹{{
                                                    formatCurrency(usersWithBalancesData.summary.total_pending_amount) }}</p>
                                            </div>
                                            <div @click="filterUsersByBalanceType('advance')"
                                                class="bg-gradient-to-br from-green-50 to-green-100 rounded-lg p-4 cursor-pointer hover:shadow-md transition border border-green-200">
                                                <p class="text-xs text-green-700 mb-1 flex items-center gap-1">
                                                    <ArrowUpCircle class="w-3 h-3" />
                                                    Total Advance
                                                </p>
                                                <p class="text-xl sm:text-2xl font-bold text-green-900">₹{{
                                                    formatCurrency(usersWithBalancesData.summary.total_advance_amount) }}</p>
                                            </div>
                                        </div>

                                        <!-- Users List -->
                                        <div class="bg-white rounded-xl shadow-sm border border-gray-200 overflow-hidden">
                                            <div class="overflow-x-auto">
                                                <table class="min-w-full divide-y divide-gray-200">
                                                    <thead class="bg-gray-50">
                                                        <tr>
                                                            <th class="px-4 py-3 text-left text-xs font-medium text-gray-500 uppercase">User</th>
                                                            <th class="px-4 py-3 text-left text-xs font-medium text-gray-500 uppercase">Balance</th>
                                                            <th class="px-4 py-3 text-left text-xs font-medium text-gray-500 uppercase">Type</th>
                                                            <th class="px-4 py-3 text-left text-xs font-medium text-gray-500 uppercase">Last Transaction</th>
                                                            <th class="px-4 py-3 text-left text-xs font-medium text-gray-500 uppercase">Actions</th>
                                                        </tr>
                                                    </thead>
                                                    <tbody class="bg-white divide-y divide-gray-200">
                                                        <tr v-for="user in usersWithBalancesData.users" :key="user.user_id"
                                                            class="hover:bg-gray-50 transition">
                                                            <td class="px-4 py-3">
                                                                <div class="text-sm font-medium text-gray-900">User #{{ user.user_id }}</div>
                                                            </td>
                                                            <td class="px-4 py-3">
                                                                <p :class="user.balance_type === 'pending' ? 'text-orange-600' : 'text-green-600'"
                                                                    class="text-sm font-bold flex items-center gap-1">
                                                                    <span v-if="user.balance_type === 'pending'">-</span>
                                                                    <span v-else>+</span>
                                                                    ₹{{ formatCurrency(user.balance_amount) }}
                                                                </p>
                                                            </td>
                                                            <td class="px-4 py-3">
                                                                <span :class="user.balance_type === 'pending' ? 'bg-orange-100 text-orange-700 border-orange-300' : 'bg-green-100 text-green-700 border-green-300'"
                                                                    class="px-2 py-1 text-xs rounded-full border font-semibold inline-flex items-center gap-1">
                                                                    <component :is="user.balance_type === 'pending' ? ArrowDownCircle : ArrowUpCircle" class="w-3 h-3" />
                                                                    {{ user.balance_type === 'pending' ? 'Pending' : 'Advance' }}
                                                                </span>
                                                            </td>
                                                            <td class="px-4 py-3">
                                                                <p class="text-xs text-gray-600">{{ formatDate(user.last_transaction_date) }}</p>
                                                            </td>
                                                            <td class="px-4 py-3 flex gap-2">
                                                                <button @click="viewUserDetails(user.user_id)"
                                                                    class="text-[#ff5100] hover:text-[#e64900] text-xs font-medium flex items-center gap-1">
                                                                    <Eye class="w-4 h-4" />
                                                                    View
                                                                </button>
                                                                <button v-if="user.balance_type === 'pending'"
                                                                    @click="openRepaymentModal(user.user_id)"
                                                                    class="text-purple-600 hover:text-purple-700 text-xs font-medium flex items-center gap-1">
                                                                    <Plus class="w-4 h-4" />
                                                                    Repay
                                                                </button>
                                                            </td>
                                                        </tr>
                                                    </tbody>
                                                </table>
